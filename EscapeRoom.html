<body>

    <h1>Escape Room Uganke</h1>

    <p>Vnesite odgovor:</p>

    <input type="text" id="answerInput" placeholder="Vaš odgovor">

    <button onclick="checkAnswer()">Preveri</button>

    <div id="message"></div>


    <script>

        // Pravilni odgovori (lahko so v katerikoli obliki velikih/malih črk)

        const correctAnswers = ["odgovor1", "odgovor2", "odgovor3", "odgovor4", "odgovor5", "odgovor6", "odgovor7"];

        let foundAnswers = [];


        function checkAnswer() {

            // Preberi vnešeni odgovor in ga pretvori v male črke

            const userAnswer = document.getElementById("answerInput").value.trim().toLowerCase();

            const messageElement = document.getElementById("message");


            // Preveri, ali je odgovor že bil pravilno vnešen

            if (foundAnswers.includes(userAnswer)) {

                messageElement.textContent = "Ta odgovor je že bil pravilno vnešen.";

                return;

            }


            // Preveri, ali je odgovor pravilen

            if (correctAnswers.includes(userAnswer)) {

                foundAnswers.push(userAnswer); // Dodaj odgovor na seznam najdenih odgovorov

                messageElement.textContent = "Pravilno! Odgovor je bil zabeležen.";

                document.getElementById("answerInput").value = ""; // Počisti polje za vnos


                // Preveri, ali so vsi odgovori pravilno vnešeni

                if (foundAnswers.length === correctAnswers.length) {

                    messageElement.textContent = "Čestitamo! Rešili ste vse uganke!";

                }

            } else {

                messageElement.textContent = "Napačen odgovor, poskusi znova!";

            }

        }

    </script>
